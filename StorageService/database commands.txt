#2 TYPE TABLE

create table TYPE (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY,
    code varchar2(20) NOT NULL,
    name varchar2(300) NOT NULL,
    PRIMARY KEY(id)
)

#3
create table SUPPLIER (
    ID NUMBER GENERATED BY DEFAULT AS IDENTITY,
    NAME varchar2(200) NOT NULL,
    PHONE_NUMBER varchar2(15),
    ADDRESS varchar2(100),
    PRIMARY KEY(ID)
)
